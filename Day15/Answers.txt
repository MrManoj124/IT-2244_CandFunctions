#include<stdio.h>
#include<unistd.h>
int main()
{
	int f=fork();
	if(f==0)
	{
		printf("I'm the child!\n");
		int sum;
		for(int i=1; i<=5; i++)
		{
			printf("%d",i);
			sum+=i;
		}
		printf("\nSummation of child Process : %d\n",sum);
	}
	else
	{
		printf("I'm the parent!\n");
		int psum;
		for(int i=6; i<=10; i++)
		{
			printf("%d",i);
			psum+=i;
		}
		printf("\nSummation of parent process: %d\n",psum);
	}
	return 0;
}

}
