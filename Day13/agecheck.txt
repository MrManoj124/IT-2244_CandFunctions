//01 - check whether you are eligible to vote.
#include<stdio.h>
int main(){
	int age;
	printf("Enter your age: ");
	scanf("%d",&age);
	if(age>0 && age<100){
		if(age>18){
		printf("You are eligible to vote!\n");
		}
		else{
			printf("You're not eligible to vote!\n");
		}
	}
	else{
		printf("Invalid age!\n");
	}
return 0;
}
