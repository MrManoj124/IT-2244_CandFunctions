Calculator

#include<stdio.h>

int main(){
	int a;
	printf("Enter a: ");
	scanf("%d", &a);
	int b;
	printf("Enter b: ");
	scanf("%d", &b);
	printf("Sum of a,b : %d\n", (a+b));
	printf("Sub of a,b : %d\n", (a-b));
	printf("Mul of a,b : %d\n", (a*b));
	printf("Div of a,b : %d\n", (a/b));
	printf("Remainder of a,b : %d\n", (a%b));
	
return 0;
}
